================================================================================

description Connection to PLDT Router 2

no shutdown
exit

network 15.15.15.0 0.0.0.3

================================================================================
                    MAIN EDGEROUTER1 (GLOBE) - FIX EXISTING CONFIG
================================================================================

enable
configure terminal

! Remove wrong EIGRP AS 200
no router eigrp 200

! Fix OSPF - remove wrong redistribution
router ospf 1
no redistribute eigrp 200
exit

! Configure correct EIGRP AS 100 for Globe
router eigrp 100
network 10.10.10.0 0.0.0.3
no auto-summary
redistribute ospf 1 metric 10000 100 255 1 1500
summary-address 192.168.0.0 255.255.0.0
exit

! Fix OSPF redistribution to use EIGRP 100
router ospf 1
redistribute eigrp 100 subnets
exit


exit
copy running-config startup-config

================================================================================
                    MAIN EDGEROUTER2 (PLDT) - FIX EXISTING CONFIG
================================================================================

enable
configure terminal

! Remove wrong EIGRP AS 100
no router eigrp 100

! Fix OSPF - remove wrong redistribution
router ospf 1
no redistribute eigrp 100
exit

! Configure correct EIGRP AS 200 for PLDT
router eigrp 200
network 11.11.11.0 0.0.0.3
no auto-summary
redistribute ospf 1 metric 10000 100 255 1 1500
summary-address 192.168.0.0 255.255.0.0
exit

! Fix OSPF redistribution to use EIGRP 200
router ospf 1
redistribute eigrp 200 subnets metric 10
exit

exit
copy running-config startup-config
